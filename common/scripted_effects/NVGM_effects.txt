nvgm_subject_investment_pool_ranking_effect = {
	clear_variable_list = nvgm_subject_investment_pool_list
	ordered_country = {
		limit = {
			OR = {
				this = root
				is_subject_of = root
			}
		}
		position = 0
		max = 1
		check_range_bounds = no
		order_by = investment_pool
		set_global_variable = {
			name = nvgm_1_subject_investment_pool_country_var
			value = investment_pool
		}
	}
	ordered_country = {
		limit = {
			OR = {
				this = root
				is_subject_of = root
			}
		}
		set_local_variable = {
			name = nvgm_subject_investment_pool_local_var
			value = investment_pool
		}
		overlord = {
			add_to_variable_list = {
				name = nvgm_subject_investment_pool_list
				target = prev
			}
		}
		order_by = investment_pool
		max = 15
		check_range_bounds = no
		change_local_variable = { 
			name = nvgm_subject_investment_pool_local_var
			divide = global_var:nvgm_1_subject_investment_pool_country_var
		}
		change_local_variable = { 
			name = nvgm_subject_investment_pool_local_var
			multiply = 550
		}
		set_variable = {
			name = nvgm_subject_investment_pool_var
			value = local_var:nvgm_subject_investment_pool_local_var
		}
	}
}
